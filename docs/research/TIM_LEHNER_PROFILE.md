## Query Used (Atlas)

```sql
-- Trace principal's full network
SELECT
    c.company_name,
    pm.state,
    COUNT(*) AS facilities,
    ROUND(AVG(mr.daily_rate), 2) AS avg_medicaid_rate
FROM principals p
JOIN principal_company_relationships pcr ON p.id = pcr.principal_id
JOIN companies c ON pcr.company_id = c.id
JOIN entities e ON c.id = e.company_id
JOIN property_entity_relationships per ON e.id = per.entity_id
JOIN property_master pm ON per.property_master_id = pm.id
LEFT JOIN medicaid_rates mr ON pm.id = mr.property_master_id AND mr.end_date IS NULL
WHERE p.full_name = 'TIMOTHY H LEHNER'
GROUP BY c.company_name, pm.state
ORDER BY c.company_name, facilities DESC;
```
