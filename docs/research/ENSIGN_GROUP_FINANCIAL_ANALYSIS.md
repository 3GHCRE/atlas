## SQL Queries Used

### Network Financial Query
```sql
SELECT
    pm.state,
    COUNT(DISTINCT pm.id) AS facilities,
    SUM(cr.total_beds) AS beds,
    FORMAT(SUM(cr.total_patient_revenue), 0) AS total_revenue,
    ROUND(AVG(mr.daily_rate), 2) AS avg_medicaid_rate
FROM companies c
JOIN entities e ON e.company_id = c.id
JOIN property_entity_relationships per ON per.entity_id = e.id
JOIN property_master pm ON pm.id = per.property_master_id
LEFT JOIN cost_reports cr ON pm.id = cr.property_master_id
    AND cr.data_source = 'cms_hcris_2024'
LEFT JOIN medicaid_rates mr ON pm.id = mr.property_master_id
    AND mr.end_date IS NULL
WHERE c.id = 7
GROUP BY pm.state
ORDER BY SUM(cr.total_patient_revenue) DESC;
```

### Quality Analysis Query
```sql
SELECT
    pm.state,
    ROUND(AVG(qr.overall_rating), 1) AS avg_overall,
    ROUND(AVG(qr.staffing_rating), 1) AS avg_staffing,
    SUM(CASE WHEN qr.overall_rating >= 4 THEN 1 ELSE 0 END) AS high_rated,
    SUM(CASE WHEN qr.overall_rating <= 2 THEN 1 ELSE 0 END) AS low_rated
FROM companies c
JOIN entities e ON e.company_id = c.id
JOIN property_entity_relationships per ON per.entity_id = e.id
JOIN property_master pm ON pm.id = per.property_master_id
LEFT JOIN quality_ratings qr ON pm.id = qr.property_master_id
WHERE c.id = 7
GROUP BY pm.state
ORDER BY AVG(qr.overall_rating) DESC;
```
