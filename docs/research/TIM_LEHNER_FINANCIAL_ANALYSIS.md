## SQL Queries Used

### Network Trace Query
```sql
-- Trace principal's full network
SELECT
    c.company_name,
    COUNT(DISTINCT e.id) AS entities,
    COUNT(DISTINCT pm.id) AS facilities,
    GROUP_CONCAT(DISTINCT pm.state ORDER BY pm.state) AS states
FROM principal_company_relationships pcr
JOIN companies c ON c.id = pcr.company_id
JOIN entities e ON e.company_id = c.id
JOIN property_entity_relationships per ON per.entity_id = e.id
JOIN property_master pm ON pm.id = per.property_master_id
WHERE pcr.principal_id = 39601
GROUP BY c.id, c.company_name
ORDER BY facilities DESC;
```

### Financial Analysis Query
```sql
-- Cost Report Data by Company
SELECT
    c.company_name,
    COUNT(DISTINCT pm.id) AS facilities,
    SUM(cr.total_beds) AS total_beds,
    FORMAT(SUM(cr.total_patient_revenue), 0) AS total_revenue,
    ROUND(AVG(cr.operating_margin) * 100, 1) AS avg_margin_pct
FROM principal_company_relationships pcr
JOIN companies c ON c.id = pcr.company_id
JOIN entities e ON e.company_id = c.id
JOIN property_entity_relationships per ON per.entity_id = e.id
JOIN property_master pm ON pm.id = per.property_master_id
LEFT JOIN cost_reports cr ON pm.id = cr.property_master_id
    AND cr.data_source = 'cms_hcris_2024'
WHERE pcr.principal_id = 39601
GROUP BY c.id, c.company_name
ORDER BY SUM(cr.total_patient_revenue) DESC;
```

### Quality & Staffing Query
```sql
-- Quality and Staffing by Company
SELECT
    c.company_name,
    ROUND(AVG(qr.overall_rating), 1) AS avg_overall,
    ROUND(AVG(sd.total_nurse_hprd), 2) AS avg_hprd,
    ROUND(AVG(sd.staff_turnover_rate), 1) AS avg_turnover
FROM principal_company_relationships pcr
JOIN companies c ON c.id = pcr.company_id
JOIN entities e ON e.company_id = c.id
JOIN property_entity_relationships per ON per.entity_id = e.id
JOIN property_master pm ON pm.id = per.property_master_id
LEFT JOIN quality_ratings qr ON pm.id = qr.property_master_id
LEFT JOIN staffing_data sd ON pm.id = sd.property_master_id
WHERE pcr.principal_id = 39601
GROUP BY c.id, c.company_name;
```
